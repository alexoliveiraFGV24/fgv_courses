#!/usr/bin/bash
#resposta_ex_5


#fiz a mesma coisa do exercício 4, mas com um loop for para colocar as informações de todos os arquivos .txt em um outro e fazer as alterações nele

touch palavras_machado.txt
for b in ~/IC/manipulando-textos-no-terminal-24-1-time_ic/Dados/machado/*/*.txt; do
    cat "$b" >> palavras_machado.txt
done
sed 's/[*** : . ; , ! ? % ( ) & " '"'"' “]/ /g' palavras_machado.txt | tr ' ' '\n' | awk '{print tolower($0)}' | sort | uniq -c | sort -n >> ~/IC/manipulando-textos-no-terminal-24-1-time_ic/Solução/resposta_ex_5
rm palavras_machado.txt
