#!/usr/bin/bash
#resposta_ex_3


#com comando awk - separa a string em 'colunas', comando sort -n - classificar numericamente

touch titulos_contos.txt titulos_contos1.txt;
for i in ~/IC/manipulando-textos-no-terminal-24-1-time_ic/Dados/machado/contos/*.txt; do
	head -qn 1 "$i" >> titulos_contos.txt;
done;
awk -F ',' '{print $3 "," $1 "," $2}' titulos_contos.txt | sort -n >> titulos_contos1.txt;
y=$(awk -F ',' '{print $2 "," $3 "," $1}' titulos_contos1.txt)
echo "$y" >> ~/IC/manipulando-textos-no-terminal-24-1-time_ic/Solução/resposta_ex_3;
rm titulos_contos.txt titulos_contos1.txt;




touch titulos_contos.txt;
for i in ~/IC/manipulando-textos-no-terminal-24-1-time_ic/Dados/machado/contos/*.txt; do
	head -qn 1 "$i" >> titulos_contos.txt;
done;
y=$(awk '{print $NF, $0}' titulos_contos.txt | sort -n | awk '{$0 = $1 " " substr($0, index($0, $2))} 1')
echo "$y" >> ~/IC/manipulando-textos-no-terminal-24-1-time_ic/Solução/resposta_ex_3;
rm titulos_contos.txt;
